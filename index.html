function checkPhysics(obj) { // --- 1. 先處理 X 軸 (水平) --- obj.x += obj.vx; // 檢查右側 if (getTile(obj.x + obj.w, obj.y + 2) || getTile(obj.x + obj.w, obj.y + obj.h - 2)) { // 撞到右邊，強制推回左邊緣外 0.1 像素 obj.x = Math.floor((obj.x + obj.w) / tileSize) * tileSize - obj.w - 0.1; obj.vx = 0; } // 檢查左側 if (getTile(obj.x, obj.y + 2) || getTile(obj.x, obj.y + obj.h - 2)) { // 撞到左邊，強制推回右邊緣外 0.1 像素 obj.x = Math.ceil(obj.x / tileSize) * tileSize + 0.1; obj.vx = 0; } // --- 2. 處理 Y 軸 (垂直) --- obj.vy += gravity; obj.y += obj.vy; obj.grounded = false; // 落地判定 (往下掉) if (obj.vy >= 0) { // 我們檢查角色的左腳與右腳 (向內縮 4 像素避免卡牆) if (getTile(obj.x + 4, obj.y + obj.h) || getTile(obj.x + obj.w - 4, obj.y + obj.h)) { // 強制將 Y 座標設定在方塊表面 obj.y = Math.floor((obj.y + obj.h) / tileSize) * tileSize - obj.h; obj.vy = 0; obj.grounded = true; } } // 撞頭判定 (往上跳) else if (obj.vy < 0) { if (getTile(obj.x + 4, obj.y) || getTile(obj.x + obj.w - 4, obj.y)) { obj.y = Math.ceil(obj.y / tileSize) * tileSize + 0.1; obj.vy = 0; } } // --- 3. 額外保險：如果角色還是嵌在土裡，強制彈出 --- if (getTile(obj.x + obj.w/2, obj.y + obj.h - 2)) { obj.y -= 2; // 微量向上彈出 } }
