function loop() { // 1. 清除畫布 ctx.clearRect(0, 0, canvas.width, canvas.height); // 2. 計算目前相機看得到的「格子範圍」 // 關鍵：startX 會隨相機移動而改變 let startX = Math.floor(cameraX / tileSize); let endX = startX + Math.ceil(canvas.width / tileSize) + 1; // 3. 只繪製這範圍內的方塊 for (let x = startX; x < endX; x++) { // y 軸通常是固定的，或者你也可以根據畫布高度動態計算 for (let y = 0; y < 25; y++) { let b = getBlock(x, y); // getBlock 內含數學公式，傳入 x 就會吐出地形 if (b !== blocks.AIR) { ctx.fillStyle = colors[b]; // 繪製位置必須減去 cameraX，才會產生「人在走、地在動」的效果 ctx.fillRect(x * tileSize - cameraX, y * tileSize, tileSize, tileSize); } } } // 4. 更新相機位置（讓它跟著玩家） // 這行沒寫的話，地圖看起來就不會延伸 cameraX = player.x - canvas.width / 2; // ... 剩下的玩家與 AI 更新 ... requestAnimationFrame(loop); }
