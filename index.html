function loop() { if(isRunning && !isPaused) { ticks = (ticks + 0.3) % 1440; let h = Math.floor(ticks/60), m = Math.floor(ticks%60); let isNight = (ticks > 1090 || ticks < 390); // 1. 處理 Y 軸移動（重力與跳躍） p.vy += 0.7; p.y += p.vy; p.ground = false; // Y 軸碰撞偵測 (地板與天花板) let ty = Math.floor(p.y / TILE); let by = Math.floor((p.y + p.h) / TILE); let lx = Math.floor((p.x + 2) / TILE); let rx = Math.floor((p.x + p.w - 2) / TILE); if (getB(lx, by) > 0 || getB(rx, by) > 0) { // 踩到地 p.y = by * TILE - p.h; p.vy = 0; p.ground = true; } else if (getB(lx, ty) > 0 || getB(rx, ty) > 0) { // 撞到天花板 p.y = (ty + 1) * TILE; p.vy = 0; } // 2. 處理 X 軸移動（左右走路） let nextVx = keys['KeyD'] ? 5 : (keys['KeyA'] ? -5 : 0); let nextX = p.x + nextVx; // X 軸碰撞偵測 (左右牆壁) let sideX = nextVx > 0 ? Math.floor((nextX + p.w) / TILE) : Math.floor(nextX / TILE); let headY = Math.floor((p.y + 2) / TILE); let footY = Math.floor((p.y + p.h - 2) / TILE); if (getB(sideX, headY) > 0 || getB(sideX, footY) > 0) { // 撞到牆壁了，不能移動 p.vx = 0; } else { p.x = nextX; p.vx = nextVx; } // 3. 跳躍 if((keys['KeyW'] || keys['Space']) && p.ground) p.vy = -12; // --- 以下維持原有的 AI 與 UI 邏輯 --- if(isNight && Math.random() < 0.01 && agents.filter(a=>a.type==='monster').length < 8) { spawnAgent('monster', p.x + (Math.random()>0.5?600:-600)); } agents.forEach((a, idx) => { a.sTimer--; if(a.sTimer <= 0) { a.dir = Math.random()>0.5?1:-1; a.sTimer = 100; } if(a.type === 'monster') a.dir = p.x > a.x ? 1 : -1; // AI 的簡單碰撞處理 (撞牆自動跳一下) let ax = a.x + a.dir * (a.type==='monster'?3:1.5); let asX = a.dir > 0 ? Math.floor((ax + a.w) / TILE) : Math.floor(ax / TILE); let ay = Math.floor((a.y + a.h - 5) / TILE); if (getB(asX, ay) > 0) { // 如果前面有方塊，自動挖掉或嘗試跳過 (這裡設為挖掉) world[asX + '_' + ay] = 0; } else { a.x = ax; } a.y += (getGY(Math.floor(a.x/TILE))*TILE - a.h - a.y) * 0.1; if(p.atk > 0 && Math.abs(a.x-p.x)<50 && Math.abs(a.y-p.y)<50) a.hp -= 10; if(a.type === 'monster' && Math.abs(a.x-p.x)<25 && Math.abs(a.y-p.y)<25) p.hp -= 0.3; if(a.hp <= 0) agents.splice(idx, 1); }); if(p.atk > 0) p.atk--; document.getElementById('posX').innerText = Math.floor(p.x/TILE); document.getElementById('posY').innerText = Math.floor(p.y/TILE); document.getElementById('hp').innerText = Math.max(0, Math.ceil(p.hp)); document.getElementById('time-text').innerText = `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}`; document.getElementById('day-icon').innerText = isNight ? "🌙" : (ticks > 1050 ? "🌅" : "☀️"); document.getElementById('biome-text').innerText = getBiomeData(Math.floor(p.x/TILE)).name; } render(); requestAnimationFrame(loop); }
